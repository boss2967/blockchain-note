1.2. 练习

  1. 数组去重

     var aList = [1,2,3,4,4,3,2,1,2,3,4,5,6,5,5,3,3,4,2,1];

 var aList2 = [];

 for(var i=0;i<aList.length;i++)
 {
     if(aList.indexOf(aList[i])==i)
     {
         aList2.push(aList[i]);
     }
 }

 alert(aList2);
